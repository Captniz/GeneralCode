          #include<stdlib.h>
          #include<time.h>
          #define DIM 10
          void carica(int []);
          void visualizza(int []);
          void bubblesort(int []);
          void selectionsort(int []);
          void insertionsort(int []);
          int main()
          {
            srand(time(NULL));
            int v[DIM];
            int c,flag;
            flag=1;
            while(1){
              printf("Scegliere un'opzione (ma prima caricare)\n\n");

          void carica(int v[DIM]){
            int i;
            for(i=0;i<DIM;i++){
              v[i]=rand()%50+1;
            }
          }
          void visualizza(int v[DIM]){
            int i;
            for(i=0;i<DIM;i++){
              printf("%3d",v[i]);
            }
            putchar('\n');
            putchar('\n');
          }
          void bubblesort(int v[DIM]){
            int i,j,box;
            for(i=0;i<DIM-1;i++){
              for(j=i+1;j<DIM;j++){
                if(v[i]>v[j]){
                  box=v[i];
                  v[i]=v[j];
                  v[j]=box;
                }
              }
            }
          }
          void selectionsort(int v[DIM]){
            int i, j, imin;
            int box;
            for(i=0; i<(DIM-1); i++){
              imin=i;
              for(j=i+1;j<DIM; j++){
                if(v[imin]>v[j]){
                  imin=j;
                }
              }
              if(i!=imin){
                box=v[i];
                v[i]=v[imin];
                v[imin]=box;
              }
            }
          }
          void insertionsort(int v[DIM]){
            int i,j,min;
            for(i=1;i<DIM;i++){
              min=v[i];
              j=i-1;
              while(min<v[j] && j>=0){
                v[j+1]=v[j];
                j=j-1;
              }
              v[j+1]=min;
            }
          }
