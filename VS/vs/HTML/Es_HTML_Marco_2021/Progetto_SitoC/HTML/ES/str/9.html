            int a,b;
            printf("inserire la base: ");

              printf("inserire l'esponente: ");
              scanf("%d",&b);

              printf("la potenza e': %li",potenza(a,b));

            }

          }
          long int potenza(int a, int b){
            long int r;
            int i;
            r=1;
            for(i=0;i<b;i++){
              r=r*a;
            }
            return r;
