          void bubblesort(int v[DIM]){
            int i,j,box;
            for(i=0;i<DIM-1;i++){
              for(j=i+1;j<DIM;j++){
                if(v[i]>v[j]){
                  box=v[i];
                  v[i]=v[j];
                  v[j]=box;
                  incSwap();
                }
                incComp();
              }
            }
          }
          void selectionsort(int v[DIM]){
            int i, j, imin;
            int box;
            for(i=0; i<(DIM-1); i++){
              imin=i;
              for(j=i+1; j<DIM; j++){
                if(v[imin]>v[j]){
                  imin=j;
                }
                incComp();
              }
              if(i!=imin){
                box=v[i];
                v[i]=v[imin];
                v[imin]=box;
                incSwap();
              }
            }
          }
          void insertionsort(int v[DIM]){
            int i,j,min;
            for(i=1;i<DIM;i++){
              min=v[i];
              j=i-1;
              while(min<v[j] && j>=0){
                v[j+1]=v[j];
                j=j-1;
                incComp();
                incSwap();
              }
              incComp();
              v[j+1]=min;
            }
          }
