            int i;
            for(i=0;i<DIM;i++){
              g[i]=rand()%50+1;
              a[i]=i+1;
              printf("%2d%3d, ",a[i],g[i]);
            }
            putchar('\n');
          }

            int i,box,box2,flag,flag2;
            do{
              flag=0;
              for(i=0;i<DIM-1;i++){
                if(g[i]<g[i+1]){
                  box=g[i];
                  g[i]=g[i+1];
                  g[i+1]=box;
                  box2=a[i];
                  a[i]=a[i+1];
                  a[i+1]=box2;

                }
              }
            }while(flag==1);
