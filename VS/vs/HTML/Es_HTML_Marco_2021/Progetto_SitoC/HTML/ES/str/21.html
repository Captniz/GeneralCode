          void space(char s[DIM]){
            int i,j;
            for(i=0;i<strlen(s);i++){
              if(s[i]==32){
                for(j=i;j<strlen(s);j++)
                  s[j]=s[j+1];			
              }
            }
          }
          void conv(char s[DIM]){
            int i;
            for(i=0;s[i]!='\0';i++){
              if(s[i]>=65 && s[i]<=90){
                s[i]=s[i]+32;
              }
            }
          }
          void sort(char s[DIM]){
            int i,j,box,flag;
            do{
              flag=0;
              for(i=0;i<strlen(s)-1;i++){
                if(s[i]>s[i+1]){
                  box=s[i];
                  s[i]=s[i+1];
                  s[i+1]=box;
                  flag=1;
                }
              }
            }while(flag==1);
